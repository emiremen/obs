CREATE
TABLE
ogrenciler
(
	id int(2) NOT NULL AUTO_INCREMENT,
    PRIMARY KEY(id),
    ogrencino int(7) NOT null,
    sinif int(1) NOT NULL,
    ayrilma_tarihi date,
    bolum_id int(2) not null,
    kimlik_id int(2) NOT null,
    adres_id int(2)
);

kimlik_id
(

    dogum_tarihi date,
    cinsiyet varchar(5) not null,

    medeni_hal varchar(5)    
);
